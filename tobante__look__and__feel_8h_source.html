56  void drawRotarySlider(Graphics& g, int x, int y, int width, int height,
57  float sliderPos, const float rotaryStartAngle,
58  const float rotaryEndAngle, Slider& slider) override
59  {
60  auto outline = slider.findColour(Slider::rotarySliderOutlineColourId);
61  auto fill = slider.findColour(Slider::rotarySliderFillColourId);
62 
63  auto bounds = Rectangle<int>(x, y, width, height).toFloat().reduced(5);
64 
65  auto radius = jmin(bounds.getWidth(), bounds.getHeight()) / 2.0f;
66  auto toAngle = rotaryStartAngle + sliderPos * (rotaryEndAngle - rotaryStartAngle);
67  auto lineW = jmin(8.0f, radius * 0.5f);
68  auto arcRadius = radius - lineW * 0.5f;
69 
70  Path backgroundArc;
71  backgroundArc.addCentredArc(bounds.getCentreX(), bounds.getCentreY(), arcRadius,
72  arcRadius, 0.0f, rotaryStartAngle, rotaryEndAngle,
73  true);
74 
75  const auto label = [&]() -> String {
76  if (slider.getName() == String("Frequency"))
77  {
78  return String(slider.getValue());
79  }
80  return String::charToString(slider.getName()[0]);
81  }();
82 
83  const auto stroke
84  = PathStrokeType(lineW, PathStrokeType::curved, PathStrokeType::rounded);
85  g.setColour(outline);
86  g.strokePath(backgroundArc, stroke);
87  g.setFont(16.0f);
88  g.drawText(label, bounds, Justification::centred, true);
89 
90  if (slider.isEnabled())
91  {
92  Path valueArc;
93  valueArc.addCentredArc(bounds.getCentreX(), bounds.getCentreY(), arcRadius,
94  arcRadius, 0.0f, rotaryStartAngle, toAngle, true);
95 
96  g.setColour(fill);
97  g.strokePath(valueArc, PathStrokeType(lineW, PathStrokeType::curved,
98  PathStrokeType::rounded));
99  }
100 
101  auto thumbWidth = lineW * 1.5f;
102  Point<float> thumbPoint(
103  bounds.getCentreX()
104  + arcRadius * std::cos(toAngle - MathConstants<float>::halfPi),
105  bounds.getCentreY()
106  + arcRadius * std::sin(toAngle - MathConstants<float>::halfPi));
107 
108  g.setColour(slider.findColour(Slider::thumbColourId));
109  g.fillEllipse(Rectangle<float>(thumbWidth, thumbWidth).withCentre(thumbPoint));
110  }

115  void drawComboBox(Graphics& g, int width, int height, bool isButtonDown, int buttonX,
116  int buttonY, int buttonW, int buttonH, ComboBox& box) override
117  {
118  ignoreUnused(buttonH);
119  ignoreUnused(buttonW);
120  ignoreUnused(buttonX);
121  ignoreUnused(buttonY);
122  ignoreUnused(isButtonDown);
123 
124  const auto* parent = box.findParentComponentOfClass<ChoicePropertyComponent>();
125  const auto cornerSize = parent != nullptr ? 0.0f : 1.0f;
126  const auto boxBounds = Rectangle<int>(0, 0, width, height);
127 
128  // Background
129  g.setColour(box.findColour(ComboBox::backgroundColourId));
130  g.fillRoundedRectangle(boxBounds.toFloat(), cornerSize);
131 
132  // Arrow
133  Path path;
134  const auto arrow = Rectangle<int>(width - 30, 0, 20, height);
135 
136  path.startNewSubPath(arrow.getX() + 3.0f, arrow.getCentreY() - 2.0f);
137  path.lineTo(static_cast<float>(arrow.getCentreX()), arrow.getCentreY() + 3.0f);
138  path.lineTo(arrow.getRight() - 3.0f, arrow.getCentreY() - 2.0f);
139 
140  const auto alpha = box.isEnabled() ? 0.9f : 0.2f;
141  g.setColour(box.findColour(ComboBox::arrowColourId).withAlpha(alpha));
142  g.strokePath(path, PathStrokeType(2.0f));
143  }

148  void getIdealPopupMenuItemSize(const String& text, bool isSeparator,
149  int standardItemHeight, int& idealWidth,
150  int& idealHeight) override
151  {
152  // This was added compared to the JUCE impl. The rest is the same
153  standardItemHeight = static_cast<int>(standardItemHeight * 1.25);
154 
155  if (isSeparator)
156  {
157  idealWidth = 50;
158  idealHeight = standardItemHeight > 0 ? standardItemHeight / 10 : 10;
159  }
160  else
161  {
162  auto font = getPopupMenuFont();
163 
164  if (standardItemHeight > 0 && font.getHeight() > standardItemHeight / 1.3f)
165  font.setHeight(standardItemHeight / 1.3f);
166 
167  idealHeight = standardItemHeight > 0 ? standardItemHeight
168  : roundToInt(font.getHeight() * 1.3f);
169  idealWidth = font.getStringWidth(text) + idealHeight * 2;
170  }
171  }
