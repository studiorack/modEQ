146  void drawRotarySlider(Graphics& g, int x, int y, int width, int height,
147  float sliderPos, const float rotaryStartAngle,
148  const float rotaryEndAngle, Slider& slider) override
149  {
150  auto outline = slider.findColour(Slider::rotarySliderOutlineColourId);
151  auto fill = slider.findColour(Slider::rotarySliderFillColourId);
152 
153  auto bounds = Rectangle<int>(x, y, width, height).toFloat().reduced(5);
154 
155  auto radius = jmin(bounds.getWidth(), bounds.getHeight()) / 2.0f;
156  auto toAngle = rotaryStartAngle + sliderPos * (rotaryEndAngle - rotaryStartAngle);
157  auto lineW = jmin(8.0f, radius * 0.5f);
158  auto arcRadius = radius - lineW * 0.5f;
159 
160  Path backgroundArc;
161  backgroundArc.addCentredArc(bounds.getCentreX(), bounds.getCentreY(), arcRadius,
162  arcRadius, 0.0f, rotaryStartAngle, rotaryEndAngle,
163  true);
164 
165  const auto label = [&]() -> String {
166  if (slider.getName() == String("Frequency"))
167  {
168  return String(slider.getValue());
169  }
170  return String::charToString(slider.getName()[0]);
171  }();
172 
173  const auto stroke
174  = PathStrokeType(lineW, PathStrokeType::curved, PathStrokeType::rounded);
175  g.setColour(outline);
176  g.strokePath(backgroundArc, stroke);
177  g.setFont(16.0f);
178  g.drawText(label, bounds, Justification::centred, true);
179 
180  if (slider.isEnabled())
181  {
182  Path valueArc;
183  valueArc.addCentredArc(bounds.getCentreX(), bounds.getCentreY(), arcRadius,
184  arcRadius, 0.0f, rotaryStartAngle, toAngle, true);
185 
186  g.setColour(fill);
187  g.strokePath(valueArc, PathStrokeType(lineW, PathStrokeType::curved,
188  PathStrokeType::rounded));
189  }
190 
191  auto thumbWidth = lineW * 1.5f;
192  Point<float> thumbPoint(
193  bounds.getCentreX()
194  + arcRadius * std::cos(toAngle - MathConstants<float>::halfPi),
195  bounds.getCentreY()
196  + arcRadius * std::sin(toAngle - MathConstants<float>::halfPi));
197 
198  g.setColour(slider.findColour(Slider::thumbColourId));
199  g.fillEllipse(Rectangle<float>(thumbWidth, thumbWidth).withCentre(thumbPoint));
200  }

205  void drawComboBox(Graphics& g, int width, int height, bool isButtonDown, int buttonX,
206  int buttonY, int buttonW, int buttonH, ComboBox& box) override
207  {
208  ignoreUnused(buttonH);
209  ignoreUnused(buttonW);
210  ignoreUnused(buttonX);
211  ignoreUnused(buttonY);
212  ignoreUnused(isButtonDown);
213 
214  const auto* parent = box.findParentComponentOfClass<ChoicePropertyComponent>();
215  const auto cornerSize = parent != nullptr ? 0.0f : 1.0f;
216  const auto boxBounds = Rectangle<int>(0, 0, width, height);
217 
218  // Background
219  g.setColour(box.findColour(ComboBox::backgroundColourId));
220  g.fillRoundedRectangle(boxBounds.toFloat(), cornerSize);
221 
222  // Arrow
223  Path path;
224  const auto arrow = Rectangle<int>(width - 30, 0, 20, height);
225 
226  path.startNewSubPath(arrow.getX() + 3.0f, arrow.getCentreY() - 2.0f);
227  path.lineTo(static_cast<float>(arrow.getCentreX()), arrow.getCentreY() + 3.0f);
228  path.lineTo(arrow.getRight() - 3.0f, arrow.getCentreY() - 2.0f);
229 
230  const auto alpha = box.isEnabled() ? 0.9f : 0.2f;
231  g.setColour(box.findColour(ComboBox::arrowColourId).withAlpha(alpha));
232  g.strokePath(path, PathStrokeType(2.0f));
233  }

238  void getIdealPopupMenuItemSize(const String& text, bool isSeparator,
239  int standardItemHeight, int& idealWidth,
240  int& idealHeight) override
241  {
242  // This was added compared to the JUCE impl. The rest is the same
243  standardItemHeight = static_cast<int>(standardItemHeight * 1.25);
244 
245  if (isSeparator)
246  {
247  idealWidth = 50;
248  idealHeight = standardItemHeight > 0 ? standardItemHeight / 10 : 10;
249  }
250  else
251  {
252  auto font = getPopupMenuFont();
253 
254  if (standardItemHeight > 0 && font.getHeight() > standardItemHeight / 1.3f)
255  font.setHeight(standardItemHeight / 1.3f);
256 
257  idealHeight = standardItemHeight > 0 ? standardItemHeight
258  : roundToInt(font.getHeight() * 1.3f);
259  idealWidth = font.getStringWidth(text) + idealHeight * 2;
260  }
261  }
