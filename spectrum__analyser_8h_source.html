41  void addAudioData(const AudioBuffer<Type>& buffer, int startChannel, int numChannels)
42  {
43  // Return if not enough space is available
44  if (abstractFifo.getFreeSpace() < buffer.getNumSamples()) { return; }
45 
46  int start1 {};
47  int block1 {};
48  int start2 {};
49  int block2 {};
50  abstractFifo.prepareToWrite(buffer.getNumSamples(), start1, block1, start2, block2);
51  audioFifo.copyFrom(0, start1, buffer.getReadPointer(startChannel), block1);
52  if (block2 > 0) { audioFifo.copyFrom(0, start2, buffer.getReadPointer(startChannel, block1), block2); }
53 
54  for (int channel = startChannel + 1; channel < startChannel + numChannels; ++channel)
55  {
56  if (block1 > 0) { audioFifo.addFrom(0, start1, buffer.getReadPointer(channel), block1); }
57  if (block2 > 0) { audioFifo.addFrom(0, start2, buffer.getReadPointer(channel, block1), block2); }
58  }
59  abstractFifo.finishedWrite(block1 + block2);
60  waitForData.signal();
61  }

74  void run() override
75  {
76  while (!threadShouldExit())
77  {
78  if (abstractFifo.getNumReady() >= fft.getSize())
79  {
80  fftBuffer.clear();
81 
82  int start1 {};
83  int block1 {};
84  int start2 {};
85  int block2 {};
86 
87  abstractFifo.prepareToRead(fft.getSize(), start1, block1, start2, block2);
88  if (block1 > 0) { fftBuffer.copyFrom(0, 0, audioFifo.getReadPointer(0, start1), block1); }
89  if (block2 > 0) { fftBuffer.copyFrom(0, block1, audioFifo.getReadPointer(0, start2), block2); }
90  abstractFifo.finishedRead(block1 + block2);
91 
92  windowing.multiplyWithWindowingTable(fftBuffer.getWritePointer(0), size_t(fft.getSize()));
93  fft.performFrequencyOnlyForwardTransform(fftBuffer.getWritePointer(0));
94 
95  ScopedLock lockedForWriting(pathCreationLock);
96  averager.addFrom(0, 0, averager.getReadPointer(averagerPtr), averager.getNumSamples(), -1.0f);
97  averager.copyFrom(averagerPtr, 0, fftBuffer.getReadPointer(0), averager.getNumSamples(),
98  1.0f / (averager.getNumSamples() * (averager.getNumChannels() - 1)));
99  averager.addFrom(0, 0, averager.getReadPointer(averagerPtr), averager.getNumSamples());
100  if (++averagerPtr == averager.getNumChannels()) { averagerPtr = 1; }
101 
102  newDataAvailable = true;
103  }
104 
105  if (abstractFifo.getNumReady() < fft.getSize()) { waitForData.wait(100); }
106  }
107  }

109  void createPath(Path& p, const Rectangle<float> bounds, float minFreq)
110  {
111  p.clear();
112 
113  ScopedLock lockedForReading(pathCreationLock);
114  const auto* fftData = averager.getReadPointer(0);
115  const auto factor = bounds.getWidth() / 10.0f;
116 
117  p.startNewSubPath(bounds.getX() + factor * indexToX(0, minFreq), binToY(fftData[0], bounds));
118  for (int i = 0; i < averager.getNumSamples(); ++i)
119  {
120  const auto x = bounds.getX() + factor * indexToX(static_cast<float>(i), minFreq);
121  const auto y = binToY(fftData[i], bounds);
122 
123  p.lineTo(static_cast<float>(x), static_cast<float>(y));
124  }
125  }

135  inline float indexToX(float index, float minFreq) const
136  {
137  const auto freq = (sampleRate * index) / fft.getSize();
138  return (freq > 0.01f) ? std::log(freq / minFreq) / std::log(2.0f) : 0.0f;
139  }
140 
141  inline float binToY(float bin, const Rectangle<float> bounds) const
142  {
143  const float infinity = -80.0f;
144  return jmap(Decibels::gainToDecibels(bin, infinity), infinity, 0.0f, bounds.getBottom(), bounds.getY());
145  }
